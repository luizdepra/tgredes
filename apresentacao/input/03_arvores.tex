\section{Árvores Multidimensionais}

\begin{frame}
	\frametitle{Árvore \textit{K-D}}
	\begin{itemize}
		\item Semelhante à árvore binária, porém possibilita a pesquisa por múltiplos atributos.
		\item Na árvore \textit{K-D} os registros são definidos por \textit{K} chaves.
		\item A chave que determina s sub-árvore que será acessada, varia a cada nível.
		\item No nível $L$ a chave $L mod(K+1)$ é utilizada.
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Árvore \textit{K-D}: Pesquisa}
	\begin{itemize}
		\item Percorre a estrutura usando os critérios escolhidos.
		\item Esses critérios diferenciam a pesquisa em exata, parcial ou por intervalo.
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Árvore \textit{K-D}: Inserção}
	\begin{itemize}
		\item Percorre a estrutura usando da pesquisa exata.
		\item Ao encontrar um nó folha, cria um nó com os novos dados e se torna filho do nó folha encontrado.
	\end{itemize}
\end{frame}
