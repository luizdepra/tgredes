\section{Árvores Multidimensionais}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
	\frametitle{Árvore \textit{K-D}}
	\begin{itemize}
		\item Semelhante à árvore binária, porém possibilita a pesquisa por múltiplos atributos.
		\item Na árvore \textit{K-D} os registros são definidos por \textit{K} chaves.
		\item A chave que determina a sub-árvore que será acessada, varia a cada nível.
		\item No nível $L$ a chave $(L\ mod\ K) + 1$ é utilizada.
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Árvore \textit{K-D}: Operações}
	\textbf{Pesquisa}
	\begin{itemize}
		\item Pode ser classificada em pesquisa exata, parcial ou por intervalo.
		\item Percorre a estrutura usando os critérios escolhidos (de acordo com o tipo de pesquisa).		
	\end{itemize}
	\textbf{Inserção}
	\begin{itemize}
		\item Percorre a estrutura usando da pesquisa exata.
		\item Ao encontrar o nó mais profundo seguindo o algoritmo da pesquisa, insere os novos dados como filho do nó encontrado.
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Árvore \textit{K-D Adaptativa}}
	\begin{itemize}
		\item Variação da árvore \textit{K-D}, resolve o problema do desbalanceamento utilizando o algoritmo de Hoare.
		\item O algoritmo encontra a mediana sobre os pontos da partição, para que a nova partição tenha o mesmo número de pontos.
		\item Funciona melhor se os dados são conhecidos a priori.
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Árvore \textit{K-D Adaptativa}: Operações}
	\textbf{Pesquisa}
	\begin{itemize}
		\item Semelhante ao algoritmo de pesquisa da árvore \textit{K-D}. Possui os mesmos tipos de pesquisa: exata, parcial e por intervalo.
		\item Porém, utiliza a mediana entre os pontos do plano atual, ao invés da chave do nó atual.
	\end{itemize}
	\textbf{Inserção}
	\begin{itemize}
		\item Segue o mesmo raciocínio da árvore \textit{K-D}. Porém, ao encontrar o nó mais profundo, calcula a mediana entre o nó a ser inserido e o nó encontrado.
		\item Cria então um nó com a mediana. Sendo esse o nó pai do nó a ser inserido e do nó encontrado.
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Árvore \textit{K-D-B}}
	\begin{itemize}
		\item Possui características de árvore \textit{K-D adaptativa} e árvore \textit{B}.
		\item Os nós internos são apontadores e representam determinada região. São chamados de \textit{páginas de regiões} e contém o endereço para o nó filho.
		\item Os dados (pontos) são armazenados somente nas \textit{páginas de pontos} (folhas).
		\item As folhas estão sempre em um mesmo nível, e isso garante que a árvore esteja balanceada.
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Árvore \textit{K-D-B}: Operações}
	\textbf{Pesquisa}
	\begin{itemize}
		\item Pode ser classificada em pesquisa exata, parcial e por intervalo.
		\item Os algoritmos de pesquisa fazem um caminho sobre a árvore até achar a página de folha, usando a divisão dos planos.
	\end{itemize}
	\textbf{Inserção}
	\begin{itemize}
		\item Para inserir um ponto $p$ na árvore, é localizada a página folha apropriada para o ponto. Se existe espaço na página folha, $p$ é inserido, caso não, a página é dividida em duas.
	\end{itemize}
\end{frame}
