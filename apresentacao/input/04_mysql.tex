\section{Banco de Dados Relacional}

\begin{frame}
	\frametitle{Banco de dados Relacional}
	\begin{itemize}
		\item Bancos de dados, são conjuntos de registros dispostos em estrutura regular que possibilitam a reorganização dos mesmos e produção de informação.
		\item Um banco de dados relacional é um conceito abstrato que define
	maneiras de armazenar, manipular e recuperar dados estruturados unicamente
	na forma de tabelas. 
		\item A tabela é um conjunto de dados dispostos em número
	finito de colunas e número ilimitado de linhas (ou tuplas).
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{SGBD Sistema Gerenciador de Bando de Dados}
	\begin{itemize}
		\item É um conjunto de programas responsável pelo gerencialmento de uma base de dados que disponibiliza uma interface para incluir, alterar ou consultar dados.
		\item Essa interface é constituída pelas APIs ou drivers do SGBD, que executam comandos na linguegem SQL
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{MySQL}
	\begin{itemize}
		\item O MySQL é um SGBD bastante popular, distribuído como
	software livre, que utiliza a linguagem SQL como interface.
	\end{itemize}
\end{frame}
