\section{Conclusão}

\begin{frame}
	\frametitle{Conclusão}
	\begin{itemize}
		\item Árvore \textit{K-D} obteve resultados melhores que o MySQL. Isso era esperado pois a árvore \textit{K-D} trabalha somente com memória primária.
		\item Árvore \textit{K-D-B} obteve inicialmente resultados melhores que o MySQL. Com ajustes no MySQL, e utilização de uma forma de inserção otimizada o resultado foi invertido.
		\item Investigações mostraram que o MySQL pode usar a árvore \textit{R} para organizar os índices. A árvore \textit{R} serve para armazenar índices espaciais.
	\end{itemize}
\end{frame}
