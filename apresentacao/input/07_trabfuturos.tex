\section{Trabalhos Futuros}

\begin{frame}
	\frametitle{Trabalhos Futuros}
	\begin{itemize}
		\item Desenvolvimento de uma interface web para a ferramenta, permitindo a configuração, monitoração e consulta multidimensional.
		\item Pode-se acoplar um módulo para geração de gráficos relativos aos recursos monitorados.
		\item Integração da ferramenta a um sistema de grade comercial com acordos de nível de serviço.
		\item Novos experimentos com a árvore \textit{R} poderiam ser realizados.
	\end{itemize}
\end{frame}
