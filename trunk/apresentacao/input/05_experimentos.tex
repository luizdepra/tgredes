\section{Experimentos}

\begin{frame}
	\frametitle{Decisões de Implementação}
	\begin{itemize}
		\item Para a medição do tempo foi utilizada a função \textit{clock()} da biblioteca \textit{time.h}.
		\item Experimento de comparação entre arquivo de entrada binário e texto.
		\item Nos arquivos de entrada texto, cada linha representa um dado multidimensional, e cada coluna uma dimensão desse dado.
		\item Número de registros variam entre 16 mil, 100 mil, 250 mil, 500 mil, 1 milhão, 2 milhões, 4 milhões e 8 milhões. 
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Comparação da árvore K-D com o MySQL}

		\item Motivação: monitoramento \textit{online}.
		\item Biblioteca libkdtree++.
		\item Registro de 4, 8 e 16 chaves.

\begin{frame}
	\frametitle{Comparação da árvore K-D-B com o MySQL: Inserção}
	\textbf{Resultados}\\
	\begin{itemize}
		\item O MySQL consegue tempos de inserção muito baixos (1 minuto e 3 segundos para 8 milhões de registros).
		\item A árvore \textit{K-D-B} fez a inserção de 8 milhões de registros em 13 minutos e 24 segundos. 
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Comparação da árvore K-D-B com o MySQL: Pesquisa}
	\textbf{Resultados}\\
	\begin{itemize}
		\item Sem índices o MySQL leva 2,8 segundos para pesquisar em uma base de 8 milhões de resitros.
		\item Para a mesma base, a árvore \textit{K-D-B} retorna o resultado em 5,6 milisegundos.
		\item O desempenho do MySQL melhora com o aumento do número de índices.
		\item Com índice de 10 chaves, o MySQL se torna mais rápido que a árvore \textit{K-D-B}.
		\item Com menos chaves, o MySQL tem um crescimento acentuado a partir de 4 milhões de registros. 
	\end{itemize}
\end{frame}
